PUT i, j, aux, X, menu as WORD
PUT A(10) as WORD

BEGIN
ET inicio
ACT:
TAKE "Tamano del arreglo" : X
LOOP UNTIL (X > 0 AND X < 11)
TRAVEL matrizObtain
TRAVEL ordernar
PRINT ("Resultados")
TRAVEL printArray
#Checar si el programa se repite
TAKE "Quiere repetir? (0)" : menu
WHEREVER menu == 0 THEN
JUMP inicio
OTHER
PRINT ("El programa ha terminado")
NOWHERE
END

SUBMETHOD ordernar:
FOR i=0 TILL X:
FOR j=i+1 TILL X:
WHEREVER A(i)>A(j) THEN
GIVE aux = A(i)
GIVE A(i) = A(j)
GIVE A(j) = aux
NOWHERE
INCOMING j
INCOMING i
RETURN

SUBMETHOD matrizObtain:
FOR i=0 TILL X:
PRINT ("Posicion") : i
TAKE "Dame el valor": A(i)
INCOMING i
RETURN

SUBMETHOD printArray:
FOR i=0 TILL X:
PRINT ("Posicion") : i
PRINT A(i)
INCOMING i
RETURN
