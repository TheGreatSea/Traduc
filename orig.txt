
PUT A(10,10) as Word
PUT B(10,10) as Word
PUT C(10,10) as Word
PUT L,P,M1,N1,M1,N1,X,Y as Word 

BEGIN
ET inicio
CLEAR
PRINT ("Dimensiones")
ACT
TAKE "Columnas M1": N1
TAKE "Renglones M1": M1
TAKE "Columnas M2": N2
TAKE "Renglones M2": M2
LOOP WHILST (N1 <> N2 AND M1 <> M2)
TRAVEL FILL_MATRIX
TRAVEL SUM_MATRIX
TRAVEL PRINT_MATRIX
TAKE "Desea repetir": L
WHEREVER L = 1 THEN
JUMP inicio
NOWHERE
END

SUBMETHOD FILL_MATRIX
PRINT ("MatrizA")
FOR X=1 TILL M1
FOR Y=1 TILL N1
PRINT("Posicion") : X, Y
TAKE "Dato": P
GIVE A(X,Y) = P
INCOMING Y
INCOMING X
PRINT ("MatrizB")
FOR X=1 TILL M1
FOR Y=1 TILL N1
PRINT("Posicion") : X, Y
TAKE "Dato": P
GIVE A(X,Y) = P
INCOMING Y
INCOMING X
RETURN


SUBMETHOD SUM_MATRIX
GIVE P = 1
GIVE L = 1
WHILST (P <= N1)
WHILST (L <= M1)
GIVE C(P,L) = A(P,L) + B(P,L)
GIVE P = P+1
GIVE L = L+1
ENDWHILST
ENDWHILST
RETURN

SUBMETHOD PRINT_MATRIX
PRINT ("Resultado")
FOR X=1 TILL M1
FOR Y=1 TILL N1
PRINT("Posicion") : X, Y
PRINT("Valor"): C(X,Y)
INCOMING Y
INCOMING X
RETURN
